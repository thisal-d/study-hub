# Solution for Question 3

```c
#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>

int main() {
    pid_t pid = fork();
    
    if (pid == 0) {
        // Child
        printf("[Child] My PID: %d, Parent PID: %d\n", getpid(), getppid());
    } else if (pid > 0) {
        // Parent
        wait(NULL);
        printf("[Parent] Child process finished execution.\n");
    }
    return 0;
}
```
