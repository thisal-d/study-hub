# Solution for Question 7

```c
#include <stdio.h>
#include <unistd.h>

int main() {
    int fd[2];
    pipe(fd);
    
    pid_t pid = fork();
    if (pid > 0) {
        // Parent
        close(fd[0]);
        int secret = 100;
        write(fd[1], &secret, sizeof(int));
        close(fd[1]);
    } else {
        // Child
        close(fd[1]);
        int val;
        read(fd[0], &val, sizeof(int));
        printf("[Child] Top Secret Code Received: %d\n", val);
        close(fd[0]);
    }
    return 0;
}
```
