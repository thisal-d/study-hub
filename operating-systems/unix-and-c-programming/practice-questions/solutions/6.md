# Solution for Question 6

```c
#include <stdio.h>
#include <pthread.h>

void* printID(void* arg) {
    int id = *(int*) arg;
    printf("Hello from Thread #%d\n", id);
    return NULL;
}

int main() {
    pthread_t threads[3];
    int ids[3] = {1, 2, 3};
    
    for (int i = 0; i < 3; i++) {
        pthread_create(&threads[i], NULL, printID, &ids[i]);
    }
    
    for (int i = 0; i < 3; i++) {
        pthread_join(threads[i], NULL);
    }
    return 0;
}
```
