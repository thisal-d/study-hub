# Solution for Question 1

```c
#include <stdio.h>

void calculateStats(int arr[], int size) {
    int sum = 0;
    int max = arr[0];
    
    for (int i = 0; i < size; i++) {
        sum += arr[i];
        if (arr[i] > max) {
            max = arr[i];
        }
    }
    
    printf("Sum: %d\n", sum);
    printf("Max: %d\n", max);
    printf("Avg: %.2f\n", (float)sum / size);
}

int main() {
    int numbers[5];
    printf("Enter 5 numbers: ");
    for (int i = 0; i < 5; i++) {
        scanf("%d", &numbers[i]);
    }
    calculateStats(numbers, 5);
    return 0;
}
```
