# Solution for Question 9

```c
#include <stdio.h>
#include <pthread.h>

void* sumArr(void* arg) {
    int *arr = (int*) arg;
    int total = 0;
    for (int i = 0; i < 4; i++) {
        total += arr[i];
    }
    printf("Array sum via thread: %d\n", total);
    return NULL;
}

int main() {
    int data[4];
    printf("Enter 4 integers: ");
    for (int i = 0; i < 4; i++) {
        scanf("%d", &data[i]);
    }
    
    pthread_t t1;
    pthread_create(&t1, NULL, sumArr, data);
    pthread_join(t1, NULL);
    return 0;
}
```
