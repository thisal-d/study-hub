# Solution for Question 8

```c
#include <stdio.h>
#include <unistd.h>

int main() {
    int fd1[2], fd2[2];
    pipe(fd1);
    pipe(fd2);
    
    pid_t pid = fork();
    if (pid > 0) {
        close(fd1[0]); close(fd2[1]);
        int num = 4;
        printf("Parent: Sending number %d...\n", num);
        write(fd1[1], &num, sizeof(int));
        
        int res;
        read(fd2[0], &res, sizeof(int));
        printf("Parent: Squared result received = %d\n", res);
    } else {
        close(fd1[1]); close(fd2[0]);
        int n;
        read(fd1[0], &n, sizeof(int));
        printf("Child: Squaring...\n");
        int square = n * n;
        write(fd2[1], &square, sizeof(int));
    }
    return 0;
}
```
