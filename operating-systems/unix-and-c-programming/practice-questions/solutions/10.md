# Solution for Question 10

```c
#include <stdio.h>
#include <unistd.h>
#include <pthread.h>
#include <sys/wait.h>

void* threadFunc(void* arg) {
    printf("Greetings from the thread!\n");
    return NULL;
}

int main() {
    pid_t pid = fork();
    if (pid == 0) {
        pthread_t t;
        pthread_create(&t, NULL, threadFunc, NULL);
        pthread_join(t, NULL);
    } else {
        wait(NULL);
        printf("Operation Complete\n");
    }
    return 0;
}
```
