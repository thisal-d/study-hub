# Solution for Question 2

```c
#include <stdio.h>

int getLength(char *ptr) {
    int count = 0;
    while (*ptr != '\0') {
        count++;
        ptr++; // Move pointer to next character
    }
    return count;
}

int main() {
    char str[100];
    printf("Enter a string: ");
    scanf("%s", str);
    
    int len = getLength(str);
    printf("Length is: %d\n", len);
    return 0;
}
```
