# Solution for Question 5

```c
#include <stdio.h>
#include <pthread.h>

typedef struct {
    int x;
    int y;
} Data;

void* multiply(void* arg) {
    Data *d = (Data*) arg;
    printf("Thread Output: Multiplication Result = %d\n", d->x * d->y);
    return NULL;
}

int main() {
    pthread_t t1;
    Data myData;
    
    printf("Enter two numbers: ");
    scanf("%d %d", &myData.x, &myData.y);
    
    pthread_create(&t1, NULL, multiply, &myData);
    pthread_join(t1, NULL);
    
    return 0;
}
```
