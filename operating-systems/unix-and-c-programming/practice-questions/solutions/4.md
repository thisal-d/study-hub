# Solution for Question 4

```c
#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>
#include <stdlib.h>

int main() {
    char path[100];
    printf("Enter command path: ");
    scanf("%s", path);
    
    pid_t pid = fork();
    if (pid == 0) {
        execl(path, path, NULL);
        perror("exec failed"); // Only runs if execl fails
        exit(1);
    } else {
        wait(NULL);
        printf("Command execution finished.\n");
    }
    return 0;
}
```
